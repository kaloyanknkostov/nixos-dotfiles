#_______________________________________________________________________yazi___________________________________________________________________________________
function y() {
	local tmp="$(mktemp -t "yazi-cwd.XXXXXX")" cwd
	yazi "$@" --cwd-file="$tmp"
	IFS= read -r -d '' cwd < "$tmp"
	[ -n "$cwd" ] && [ "$cwd" != "$PWD" ] && builtin cd -- "$cwd"
	rm -f -- "$tmp"
}
#_______________________________________________________________________fzf around_____________________________________________________________________________
f() {
  # Get the list of most-used dirs from zoxide and pipe to fzf
  local dir
  dir=$(zoxide query -l | fzf --height 40%)

  # If 'dir' is not empty, cd to it
  if [[ -n "$dir" ]]; then
    cd "$dir"
  fi
}
